		; 50 - 30
		push    50
		push    30
		sub

		; 5 - 3
		push    5
		push    3
		sub

		div			; (50 - 30) / (5 - 3)

		push 5
		call print_ntimes
		halt


print_ntimes:	pop rcx			; initialize loop counter
		pop rax			; store data

p_loop_start:	push rcx
		push 0
		jle p_loop_end		; check exit condition

		push rax
		out			; print number

		push rcx		; decrement counter
		dec
		pop rcx

		jmp p_loop_start	; repeat loop
p_loop_end:	ret
